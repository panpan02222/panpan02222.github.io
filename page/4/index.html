    wget https://github.com/git-lfs/git-lfs/releases/download/v3.1.2/git-lfs-linux-arm64-v3.1.2.tar.gz
    tar -xvf git-lfs-linux-arm64-v3.1.2.tar.gz

    sudo ./install.sh

    git lfs install
